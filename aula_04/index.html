using namespace std;

int main() {
    double a, b, c;
    
    cout << "Digite os coeficientes a, b e c: ";
    cin >> a >> b >> c;

    if (a == 0) {
        cout << "Isso nao e uma equacao do segundo grau." << endl;
        return 0;
    }

    double delta = b * b - 4 * a * c;

    if (delta > 0) {
        double x1 = (-b + sqrt(delta)) / (2 * a);
        double x2 = (-b - sqrt(delta)) / (2 * a);
        cout << "Duas raizes reais:" << endl;
        cout << "x1 = " << x1 << endl;
        cout << "x2 = " << x2 << endl;
    } 
    else if (delta == 0) {
        double x = -b / (2 * a);
        cout << "Uma raiz real (raiz dupla):" << endl;
        cout << "x = " << x << endl;
    } 
    else {
        double real = -b / (2 * a);
        double imag = sqrt(-delta) / (2 * a);
        cout << "Raizes complexas:" << endl;
        cout << "x1 = " << real << " + " << imag << "i" << endl;
        cout << "x2 = " << real << " - " << imag << "i" << endl;
    }

    return 0;
}
